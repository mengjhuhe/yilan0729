        // Gemini API integration

        async function callGeminiAPI(prompt, displayElementId, loadingElementId) {

            const displayElement = document.getElementById(displayElementId);

            const loadingElement = document.getElementById(loadingElementId);


            displayElement.classList.add('hidden');

            loadingElement.classList.remove('hidden');

            displayElement.innerHTML = ''; // Clear previous content


            try {

                let chatHistory = [];

                chatHistory.push({ role: "user", parts: [{ text: prompt }] });

                const payload = { contents: chatHistory };

                const apiKey = ""; // If you want to use models other than gemini-2.0-flash or imagen-3.0-generate-002, provide an API key here. Otherwise, leave this as-is.

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });


                const result = await response.json();


                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    const text = result.candidates[0].content.parts[0].text;

                    displayElement.innerHTML = text.replace(/\n/g, '<br>'); // Display newlines as breaks

                } else {

                    displayElement.innerHTML = '無法取得建議，請稍後再試。';

                }

            } catch (error) {

                console.error('Error calling Gemini API:', error);

                displayElement.innerHTML = '取得建議時發生錯誤，請檢查網路連線或稍後再試。';

            } finally {

                loadingElement.classList.add('hidden');

                displayElement.classList.remove('hidden');

            }

        }
